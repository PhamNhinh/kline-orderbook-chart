Orderbook heatmap in 15 lines

This is all it takes to get candlesticks with a live orderbook heatmap.

app.js 
import { MrdChart } from '@mrd/chart-engine'

const chart = await MrdChart.create(canvas, {
  licenseKey: 'MRD-PRO-XXXXXXXX-20270101-XXXXXX',
  theme: 'dark',
})

chart.setKlines(ohlcvData)
chart.setHeatmap({ data: depthMatrix, rows: 200, cols: 100,
  priceMin: 64000, priceMax: 66000,
  timeStart: 1710000000, timeEnd: 1710050000,
})
chart.indicators.enable('footprint')
chart.start()

// Real-time updates
ws.on('kline', k => chart.updateLastKline(k))
ws.on('depth', col => chart.updateHeatmapColumn(col.i, col.data))
